> **Summary:** This application uses Canvas, animation functions, and XSJS services to control moving squares. Information is saved to and retrieved from HANA based on the current positions of the squares, and the blue square is driven by configuration stored in HANA tables.

## Introduction

This application uses Canvas, animation functions, and XSJS services to control the movement of squares.

Information is saved to and retrieved from HANA based on the current positions of the squares.

The blue square in the demo is controlled by configuration data stored in HANA tables.

window.requestAnimFrame = (function(callback) {
  return window.requestAnimationFrame ||
    window.webkitRequestAnimationFrame ||
    window.mozRequestAnimationFrame ||
    window.oRequestAnimationFrame ||
    window.msRequestAnimationFrame ||
    function(callback) {
      window.setTimeout(callback, 1000 / 60);
    };

function drawRectangle(myRectangle, context) {
  context.beginPath();
  context.rect(myRectangle.x, myRectangle.y, myRectangle.width, myRectangle.height);
  context.fillStyle = '#8ED6FF';
  context.fill();
  context.lineWidth = myRectangle.borderWidth;
  context.strokeStyle = 'black';
  context.stroke();
}

function animate(myRectangle, canvas, context, startTime) {
  // The numbers below are the x/y coordinates of the squares.
  // These values can be retrieved dynamically from HANA.
  // The condition defines the speed, direction, and position of the square.
  // Similar logic is repeated for square 2.
  if (myRectangle.x >= 145 && myRectangle.x <= 1116 && myRectangle.y >= 100 && myRectangle.y <= 233) {
    var spd = oResult.getText();
    myRectangle.x = Number(myRectangle.x + (spd * 1));

    $(function(oEvent) {
      if (spd == 'null' || spd == null || spd.length === 0 || typeof spd === "undefined") {
        var Movobj = 'SQ1';
        oController.onLiveRace1(Movobj);
      }
    });
  }

  // Additional movement logic continues here...

  requestAnimFrame(function() {
    animate2(myRectangle2, canvas, context, startTime);
  });
}

var canvas = document.getElementById('myCanvas');
var context = canvas.getContext('2d');
var myRectangle = { x: 145, y: 100, width: 12, height: 12, borderWidth: 1 };

**.XSJS**

Below are sample snippets to retrieve data by:

- calling a procedure
- selecting from a table

```javascript
function performSpeed() {
  var body = '';
  var x1 = $.request.parameters.get('x1');
  var y1 = $.request.parameters.get('y1');
  var x2 = $.request.parameters.get('x2');
  var y2 = $.request.parameters.get('y2');

  try {
    var query = 'CALL "amohas97.session.models::get_speed"(?,?,?,?,? )';
    $.trace.debug(query);
    var conn = $.db.getConnection();
    var cstmt = conn.prepareCall(query);
    cstmt.setString(1, x1);
    cstmt.setString(2, y1);
    cstmt.setString(3, x2);
    cstmt.setString(4, y2);
    cstmt.execute();
    body += cstmt.getDecimal(5) + "\n";
  } catch (e) {
    $.response.status = $.net.http.INTERNAL_SERVER_ERROR;
    $.response.setBody(e.message);
    return;
  }

  $.response.setBody(body);
  $.response.status = $.net.http.OK;
}

function performRace1() {
  var body = '';
  var Movobj = $.request.parameters.get('Movobj');

  try {
    var query = 'select "Defaults" FROM "WRK_SCH"."amohas97.session.data::sdyn" WHERE "Movobj" = ? ';
    $.trace.debug(query);
    var conn = $.db.getConnection();
    var pstmt = conn.prepareStatement(query);
    pstmt.setString(1, Movobj);
    var rs = pstmt.executeQuery();

    while (rs.next()) {
      body += rs.getNString(1) + "\n";
    }
  } catch (e) {
    $.response.status = $.net.http.INTERNAL_SERVER_ERROR;
    $.response.setBody(e.message);
    return;
  }

  $.response.setBody(body);
  $.response.status = $.net.http.OK;
}
```
